举个栗子
```javascript
var str = 'hahaha123'
console.log(str.includes('1'));     //true
console.log(str.indexOf('1'));      //6
var arr = [1,2,3,4,5]
console.log(arr.includes(1));       //true
console.log(arr.indexOf(1));        //0
```
